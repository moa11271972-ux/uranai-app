function sumDigits(text){
  return String(text).replace(/[^0-9]/g,'').split('').reduce((sum,n)=>sum+Number(n),0);
}
function reduce(num){
  let n = num;
  while(n > 9){
    n = String(n).split('').reduce((a,b)=>a+Number(b),0);
  }
  return n;
}

function getKyusei(birthdate){
  const [y,m,d] = birthdate.split('-').map(Number);
  let year = y;
  if(m < 2 || (m === 2 && d < 4)) year = y - 1;
  let num = 11 - (year % 9);
  while(num > 9) num -= 9;
  while(num < 1) num += 9;
  return kyuseiMap[num];
}

function setQuickTheme(value, el){
  document.getElementById('theme').value = value;
  document.querySelectorAll('.chip').forEach(chip => chip.classList.remove('active'));
  el.classList.add('active');
}
function showError(msg){
  const box = document.getElementById('errorBox');
  box.textContent = msg;
  box.style.display = 'block';
  document.getElementById('okBox').style.display = 'none';
}
function showOk(msg){
  const box = document.getElementById('okBox');
  box.textContent = msg;
  box.style.display = 'block';
  document.getElementById('errorBox').style.display = 'none';
}
function hideStatus(){
  document.getElementById('errorBox').style.display = 'none';
  document.getElementById('okBox').style.display = 'none';
}
function diagnose(){
  hideStatus();
  const name = document.getElementById('name').value.trim();
  const gender = document.getElementById('gender').value;
  const birthdate = document.getElementById('birthdate').value;
  const birthtime = document.getElementById('birthtime').value || '12:00';
  const theme = document.getElementById('theme').value;
  const focus = document.getElementById('focus').value;
  const worry = document.getElementById('worry').value.trim();
  const partnerBirth = document.getElementById('partnerBirth').value;
  const moveTiming = document.getElementById('moveTiming').value.trim();
  const moveDirection = document.getElementById('moveDirection').value.trim();
  const readerMemo = document.getElementById('readerMemo').value.trim();

  if(!birthdate){
    showError('生年月日を入力してください。');
    return;
  }

  const result = buildIntegratedAnswer({
    name:name || 'あなた', gender, birthdate, birthtime, theme, focus, worry, partnerBirth, moveTiming, moveDirection, readerMemo
  });

  document.getElementById('resultBox').textContent = result.main;
  document.getElementById('lifeText').textContent = result.life;
  document.getElementById('pillarText').textContent = result.pillar;
  document.getElementById('ekiText').textContent = result.eki;
  document.getElementById('colorText').textContent = result.color;
  document.getElementById('kyuseiText').textContent = result.kyusei;
  document.getElementById('astroText').textContent = result.astro;
  document.getElementById('numText').textContent = result.num;
  document.getElementById('sanmeiText').textContent = result.sanmei;
  document.getElementById('nameText').textContent = result.name;
  document.getElementById('combinedText').textContent = result.combined;
  document.getElementById('actionText').textContent = result.action;
  document.getElementById('themeText').textContent = result.theme;
  document.getElementById('report1').textContent = result.report1;
  document.getElementById('report2').textContent = result.report2;

  document.getElementById('resultBox').classList.remove('hidden');
  document.getElementById('detailGrid').classList.remove('hidden');
  document.getElementById('reportCard').classList.remove('hidden');
  showOk('鑑定結果を表示しました。');
}
function copyResult(){
  const result = document.getElementById('resultBox');
  if(result.classList.contains('hidden')){
    showError('先に鑑定結果を表示してください。');
    return;
  }
  const text = `${result.textContent}\n\n${document.getElementById('report1').textContent}\n\n${document.getElementById('report2').textContent}`;
  navigator.clipboard.writeText(text).then(()=>showOk('結果をコピーしました。')).catch(()=>showError('コピーに失敗しました。'));
}
function resetForm(){
  ['name','worry','moveTiming','moveDirection','readerMemo'].forEach(id => document.getElementById(id).value = '');
  document.getElementById('gender').value = '女性';
  document.getElementById('birthdate').value = '';
  document.getElementById('birthtime').value = '12:00';
  document.getElementById('theme').value = '総合運';
  document.getElementById('focus').value = '動くべきか';
  document.getElementById('partnerBirth').value = '';
  document.querySelectorAll('.chip').forEach(chip => chip.classList.remove('active'));
  document.getElementById('resultBox').classList.add('hidden');
  document.getElementById('detailGrid').classList.add('hidden');
  document.getElementById('reportCard').classList.add('hidden');
  hideStatus();
}
function saveInput(){
  const data = {
    name:document.getElementById('name').value,
    gender:document.getElementById('gender').value,
    birthdate:document.getElementById('birthdate').value,
    birthtime:document.getElementById('birthtime').value,
    theme:document.getElementById('theme').value,
    focus:document.getElementById('focus').value,
    worry:document.getElementById('worry').value,
    partnerBirth:document.getElementById('partnerBirth').value,
    moveTiming:document.getElementById('moveTiming').value,
    moveDirection:document.getElementById('moveDirection').value,
    readerMemo:document.getElementById('readerMemo').value
  };
  localStorage.setItem('moa_integrated_fortune_input', JSON.stringify(data));
  showOk('入力内容を保存しました。');
}
function loadInput(){
  const raw = localStorage.getItem('moa_integrated_fortune_input');
  if(!raw){
    showError('保存された入力がありません。');
    return;
  }
  const data = JSON.parse(raw);
  Object.keys(data).forEach(key => {
    const el = document.getElementById(key);
    if(el) el.value = data[key] || '';
  });
  showOk('保存した入力を復元しました。');
}
