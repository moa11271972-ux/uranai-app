    function buildPerson(name, birth, birthTime){
    const [y,m,d] = birth.split("-").map(Number);
    const hour = birthTime ? Number(birthTime.split(":")[0]) : 12;
    return {
    name,
    y,m,d,hour,
    sun:getSunSign(m,d),
    moon:getMoonSign(m),
    asc:getAscSign(hour),
    lifePath:getLifePath(birth)
    };
    }
    function getTarot(seed){
    return rand(tarotCards, seed);
    }
    function getLenormand(seed){
    return rand(lenormandCards, seed + "len");
    }
    function getRune(seed){
    return rand(runes, seed + "rune");
    }
    function getCompatibility(person, partner){
    let score = 0;
    let text = [];
    if(person.sun === partner.sun){
    score += 35;
    text.push("太陽星座が同じため、根本的な感覚や価値観が通じやすい相性です。");
    } else {
    text.push("太陽星座が異なるため、違いを理解すると補い合いやすい関係です。");
    }
    if(person.lifePath === partner.lifePath){
    score += 30;
    text.push("数秘のライフパスが同じため、人生テーマが似ています。");
    }
    if(person.moon === partner.moon){
    score += 25;
    text.push("月の性質が近く、感情面で理解し合いやすいです。");
    }
    const label = score >= 70 ? "非常に相性が良い" : score >= 40 ? "学び合える良い相性" : "努力で深まる相性";
    return `${partner.name}様との相性判定：${label}
    ${text.join("\n")}`;
    }

    function generateReading(forcePremium){
    const name = document.getElementById("name").value.trim();
    const birth = document.getElementById("birth").value;
    const birthTime = document.getElementById("birthTime").value;
    const partnerName = document.getElementById("partnerName").value.trim();
    const partnerBirth = document.getElementById("partnerBirth").value;
    const partnerBirthTime = document.getElementById("partnerBirthTime").value;
    const question = document.getElementById("question").value.trim();
    if(!name || !birth){
    alert("お名前と生年月日を入力してください");
    return;
    }
    const seed = name + birth + question + currentMode;
    const person = buildPerson(name, birth, birthTime);
    const tarot = getTarot(seed);
    const lenormand = getLenormand(seed);
    const rune = getRune(seed);
    let compatibilityText = "お相手の情報が未入力です。";
    if(partnerName && partnerBirth){
    const partner = buildPerson(partnerName, partnerBirth, partnerBirthTime);
    compatibilityText = getCompatibility(person, partner);
    }
    const preview = buildPreview(person, tarot, rune);
    const detail = buildDetail(person, tarot, lenormand, rune);
    const story = premiumUnlocked || forcePremium
    ? createStory(person, question, currentMode, tarot, lenormand, rune, compatibilityText)
    : "ここから先は有料版で表示される物語鑑定です。";
    const fullText = `${preview}
    ${detail}
    【相性】
    ${compatibilityText}
    【有料版 物語鑑定】
    ${story}`;
    document.getElementById("plainResult").innerText = fullText;
    document.getElementById("reportPreview").innerText = preview;
    document.getElementById("reportDetail").innerText = `${detail}
    【相性】
    ${compatibilityText}`;
    document.getElementById("reportStory").innerText = story;
    document.getElementById("reportView").classList.remove("hidden");
    document.getElementById("planLabel").innerText = premiumUnlocked || forcePremium ? "現在：有料版デモ表示" : "現在：無料版プレビュー";
    if(!(premiumUnlocked || forcePremium)) showPremiumLock();
    }

    function saveHistory(){
    const text = document.getElementById("plainResult").innerText;
    if(!text){
    alert("先に鑑定してください");
    return;
    }
    const history = JSON.parse(localStorage.getItem("western_premium_history") || "[]");
    history.unshift({
    createdAt:new Date().toLocaleString("ja-JP"),
    text
    });
    localStorage.setItem("western_premium_history", JSON.stringify(history.slice(0,20)));
    alert("履歴を保存しました");
    }

    document.getElementById("reportStory").innerText = "";
    }
